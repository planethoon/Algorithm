function solution(board, moves) {
  let queue = [];
  return moves.reduce((acc, cur) => {
    for (let elem of board) {
      if (elem[cur - 1]) {
        if (queue[queue.length - 1] !== elem[cur - 1]) {
          queue.push(elem[cur - 1]);
          elem[cur - 1] = 0;
          return acc;
        } else {
          queue.pop();
          elem[cur - 1] = 0;
          return acc + 2;
        }
      }
    }
    return acc;
  }, 0);
}
