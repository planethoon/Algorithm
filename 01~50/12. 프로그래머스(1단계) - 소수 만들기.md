function solution(nums) {
    let countPrime = 0;
    const isPrime = (num) => {
        let count = 0;
        for(let i = 0; i <= num; i++){
            if(num % i === 0){
                count++
            }
        }
        if(count === 2){
            return true;
        }
        return false;
    }
    
    for(let i = 0; i < nums.length-2; i++){
        for(let j = i+1; j < nums.length-1; j++){
            for(let k = j+1; k < nums.length; k++){
                if(isPrime(nums[i]+nums[j]+nums[k])){
                    countPrime++;
                }
            }
        }
    }
    
    
    return countPrime;
}
